261 void base_island::int_evolver::juice_impl(boost::posix_time::ptime &start)
262 {
263  start = boost::posix_time::microsec_clock::local_time();
264  // Synchronise start with all other threads if we are in an archi.
265  if (m_i->m_archi) {
266  m_i->m_archi->sync_island_start();
267  }
268  const raii_thread_hook hook(m_i);
269  for (std::size_t i = 0; i < m_n; ++i) {
270  // Call pre-evolve hooks.
271  if (m_i->m_archi) {
272  m_i->m_archi->pre_evolution(*m_i);
273  }
274  m_i->m_pop.problem().pre_evolution(m_i->m_pop);
275  // Call the evolution.
276  m_i->perform_evolution(*m_i->m_algo,m_i->m_pop);
277  // Post-evolve hooks.
278  if (m_i->m_archi) {
279  m_i->m_archi->post_evolution(*m_i);
280  }
281  m_i->m_pop.problem().post_evolution(m_i->m_pop);
282  // Set the interruption point.
283  boost::this_thread::interruption_point();
284  }
285 }

287 // Implementation of int_evolver's juice.
288 void base_island::int_evolver::operator()()
289 {
290  boost::posix_time::ptime start;
291  try {
292  juice_impl(start);
293  } catch (const boost::thread_interrupted &) {
294  // In case of interruption, don't do anything special.
295  } catch (const std::exception &e) {
296  std::cout << "Error during island evolution using " << m_i->m_algo->get_name() << ": " << e.what() << std::endl;
297  } catch (...) {
298  std::cout << "Error during island evolution using " << m_i->m_algo->get_name() << ", unknown exception caught. :(" << std::endl;
299  }
300  // Try to compute the evolution time before exiting. In case something goes wrong, do not do anything.
301  try {
302  // We must take care of potentially low-accuracy clocks, where the time difference could be negative for
303  // _really_ short evolution times. In that case do not add anything to the total evolution time.
304  const boost::posix_time::time_duration diff = boost::posix_time::microsec_clock::local_time() - start;
305  if (diff.total_milliseconds() >= 0) {
306  m_i->m_evo_time += boost::numeric_cast<std::size_t>(diff.total_milliseconds());
307  }
308  } catch (...) {
309  std::cout << "Error calculating evolution time.\n";
310  }
311 }

344 void base_island::t_evolver::juice_impl(boost::posix_time::ptime &start)
345 {
346  boost::posix_time::time_duration diff;
347  start = boost::posix_time::microsec_clock::local_time();
348  // Synchronise start.
349  if (m_i->m_archi) {
350  m_i->m_archi->sync_island_start();
351  }
352  const raii_thread_hook hook(m_i);
353  do {
354  if (m_i->m_archi) {
355  m_i->m_archi->pre_evolution(*m_i);
356  }
357  m_i->m_pop.problem().pre_evolution(m_i->m_pop);
358  m_i->perform_evolution(*m_i->m_algo,m_i->m_pop);
359  if (m_i->m_archi) {
360  m_i->m_archi->post_evolution(*m_i);
361  }
362  m_i->m_pop.problem().post_evolution(m_i->m_pop);
363  // Set the interruption point.
364  boost::this_thread::interruption_point();
365  diff = boost::posix_time::microsec_clock::local_time() - start;
366  // Take care of negative timings.
367  } while (diff.total_milliseconds() < 0 || boost::numeric_cast<std::size_t>(diff.total_milliseconds()) < m_t);
368 }

582 // Accept individuals incoming from a migration operation. Returns an std::vector containing the number of accepted individuals from each island
583 std::vector<std::pair<population::size_type, archipelago::size_type> > base_island::accept_immigrants(std::vector<std::pair<population::size_type, population::individual_type> > &immigrant_pairs)
584 {
585  std::vector<std::pair<population::size_type, archipelago::size_type> > retval;
586  // Make sure we are in an archipelago.
587  pagmo_assert(m_archi);
588  // We shuffle the immigrants as to make sure not to give preference to a particular island
589  boost::uniform_int<int> pop_idx(0,immigrant_pairs.size());
590  boost::variate_generator<boost::mt19937 &, boost::uniform_int<int> > p_idx(m_pop.m_urng,pop_idx);
591  std::random_shuffle(immigrant_pairs.begin(),immigrant_pairs.end(), p_idx);
592  // We extract the immigrants from the pair
593  std::vector<population::individual_type> immigrants;
594  immigrants.reserve(immigrant_pairs.size());
595  for (size_t i=0;i<immigrant_pairs.size();++i) {
596  immigrants.push_back(immigrant_pairs[i].second);
597  }
598 
599  std::vector<std::pair<population::size_type,std::vector<population::individual_type>::size_type> > rep;
600  rep = m_r_policy->select(immigrants,m_pop);
601  for (std::vector<std::pair<population::size_type,std::vector<population::individual_type>::size_type> >::const_iterator
602  rep_it = rep.begin(); rep_it != rep.end(); ++rep_it)
603  {
604  pagmo_assert((*rep_it).first < m_pop.m_container.size() && (*rep_it).second < immigrants.size());
605  m_pop.m_container[(*rep_it).first] = immigrants[(*rep_it).second];
606  m_pop.update_champion((*rep_it).first);
607  m_pop.update_dom((*rep_it).first);
608  std::pair<population::size_type, archipelago::size_type> pair = std::make_pair(1.0, immigrant_pairs[(*rep_it).second].first);
609  std::vector<std::pair<population::size_type, archipelago::size_type> >::iterator where;
610  where = std::find_if(retval.begin(), retval.end(), unary_predicate(pair));
611  if (where == retval.end()) {
612  retval.push_back(pair);
613  }
614  else {
615  (*where).first++;
616  }
617  }
618  return(retval);
619 }
