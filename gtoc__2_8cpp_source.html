59 gtoc_2::gtoc_2(int ast1, int ast2, int ast3, int ast4, int n_seg, objective obj):
60  base(12 * n_seg + 15,0,1,7 * 4 + n_seg * 4 + 1, n_seg * 4 + 1, 1E-3),
61  m_n_seg(n_seg),m_spacecraft(1500.,.1,4000.),m_obj(obj)
62 {
63  if (n_seg <= 0) {
64  pagmo_throw(value_error,"invalid number of segments");
65  }
66  m_asteroids.push_back(asteroid_gtoc2(910));

69  m_asteroids.push_back(asteroid_gtoc2(ast3));

73  m_legs.push_back(leg());
74  m_legs.back().set_spacecraft(m_spacecraft);
75  m_legs.back().set_mu(1.32712440018e20);

78  // Check that input asteroids belong to different groups.

80  m_asteroids[3].get_group(), m_asteroids[4].get_group()};
81  std::sort(asteroid_groups,asteroid_groups + 4);
82  if (std::unique(asteroid_groups,asteroid_groups + 4) - asteroid_groups != 4) {
83  pagmo_throw(value_error,"asteroids must belong to different groups");
84  }
85  if (std::find(asteroid_groups,asteroid_groups + 4,5) != asteroid_groups + 4) {
86  pagmo_throw(value_error,"the Earth cannot appear in the list of asteroids");
87  }
88  decision_vector lb_v, ub_v;
89  // Launch date.
90  lb_v.push_back(57023.5);
91  ub_v.push_back(64693.5);
92  // Flight and waiting times.
93  for (int i = 0; i < 3; ++i) {
94  // Flight time.
95  lb_v.push_back(5);
96  ub_v.push_back(3600);
97  // Waiting time.

99  ub_v.push_back(90.0000001);
100  }

102  lb_v.push_back(5);
103  ub_v.push_back(3600);

106  lb_v.push_back(100);
107  ub_v.push_back(1500);
108  }

127 void gtoc_2::objfun_impl(fitness_vector &f, const decision_vector &x) const
128 {
129  // Objective function is m_f / t_f.
130  switch (m_obj){
131  case MASS:
132  f[0] = - x[11];
133  break;
134  case TIME:
135  f[0] = (std::accumulate(x.begin() + 1,x.begin() + 8,0.) * ASTRO_DAY2YEAR);
136  break;
137  case MASS_TIME:
138  f[0] = - x[11] / (std::accumulate(x.begin() + 1,x.begin() + 8,0.) * ASTRO_DAY2YEAR);
139  }

142 void gtoc_2::compute_constraints_impl(constraint_vector &c, const decision_vector &x) const
143 {
144  // Cached values.
145  array3D r, v;
146  double initial_mass = m_spacecraft.get_mass();
147  double final_mass = x[8];
148  // Build legs.
149  for (int i = 0; i < 4; ++i) {
150  // Calculate start-end leg epochs.
151  epoch start(std::accumulate(x.begin(),x.begin() + 2 * i + 1, 0.),epoch::MJD),
152  end(std::accumulate(x.begin(),x.begin() + 2 * i + 2, 0.),epoch::MJD);
153  // Set leg's start-end epochs.
154  m_legs[i].set_t_i(start);
155  m_legs[i].set_t_f(end);
156  // Initial state.
157  m_asteroids[i].get_eph(start,r,v);
158  // First leg: Vinf is added to the state.
159  if (i == 0) {
160  for (int j = 0; j < 3; ++j) {
161  v[j] += x[12 + j] * 1000;
162  }
163  }
164  m_legs[i].set_x_i(sc_state(r,v,initial_mass));
165  // Throttles.
166  for (int j = 0; j < m_n_seg; ++j) {
167  m_legs[i].set_throttles(j,get_nth_throttle(j,x.begin() + 15 + 3 * i * m_n_seg,start,end));
168  }
169  // Final state.
170  m_asteroids[i + 1].get_eph(end,r,v);
171  m_legs[i].set_x_f(sc_state(r,v,final_mass));
172  // Update masses.
173  initial_mass = final_mass;
174  // Do not update mass if final iteration (useless).
175  if (i != 3) {
176  final_mass = x[9 + i];
177  }
178  }
179  // Load state mismatches into constraints vector.
180  for (int i = 0; i < 4; ++i) {
181  m_legs[i].get_mismatch_con(c.begin() + 7 * i, c.begin() + 7 * (i + 1));
182  // Passing non-dimensional units to the solver.
183  for (int j = 0; j < 3; ++j) {
184  c[7 * i + j] /= ASTRO_AU;
185  c[7 * i + j + 3] /= ASTRO_EARTH_VELOCITY;
186  }
187  c[7*i+6] /=1500;
188  }
189  // Throttles constraints.
190  for (int i = 0; i < 4; ++i) {
191  m_legs[i].get_throttles_con(c.begin() + 28 + i * m_n_seg, c.begin() + 28 + (i + 1) * m_n_seg);
192  }
193  // Vinf constraint.
194  c.back() = (x[12] * x[12] + x[13] * x[13] + x[14] * x[14] - 3.5 * 3.5) / (ASTRO_EARTH_VELOCITY * ASTRO_EARTH_VELOCITY) * 1000000;
195 }
