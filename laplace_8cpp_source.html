36 static inline int laplace_get_dimension(const std::vector<int> &seq)
37 {
38  if (seq.size() < 2)
39  {
40  pagmo_throw(value_error,"fly-by sequence must contain at least two planets!! Earth and Jupiter");
41  }
42  if (seq[0]!=3)
43  {
44  pagmo_throw(value_error,"Starting planet must be the Earth (3) for the Laplace mission");
45  }
46  if (seq.back()!=5)
47  {
48  pagmo_throw(value_error,"Final planet must be Jupiter (5) for the Laplace mission");
49  }
50  for (size_t i=0; i< seq.size(); ++i) {
51  if ((seq[i] >= 7) || (seq[i] <= 0)) {
52  pagmo_throw(value_error,"Planet sequence must contains numbers from 1 to 6");
53  }
54  }
55  return boost::numeric_cast<int>(4*seq.size() - 2);
56 }

107 void laplace::objfun_impl(fitness_vector &f, const decision_vector &x) const
108 {
109  f[0] = 0;
110  MGA_DSM(x, problem, f[0]);
111  const size_t sequence_size = (x.size() + 2) / 4;
112  double totaltime = 0;
113  for (size_t i = 0; i < sequence_size - 1 ; ++i) {
114  totaltime += x[i + 4];
115  }
116  const double delta = totaltime - 8 * 365.25;
117  // Penalise trajectory longer than 8 years by 200 meters/s per month.
118  f[0] = std::max<double>(f[0],f[0] + 0.2 / 30 * delta);
119 }

131  std::string laplace::pretty(const std::vector<double> &x) const
132 {
133  double obj = 0;

135  std::ostringstream s;

139  pagmo_assert((x.size() + 2) % 4 == 0 && seq_size >= 2);
140  pagmo_assert(problem.sequence.size() == seq_size);
141  s << "Flyby sequence: ";
142  for (size_t i = 0; i < seq_size; ++i) {
143  s << problem.sequence[i];
144  }
145  s << '\n';

148  s << "Vinf polar components: ";
149  for (size_t i = 0; i < 3; ++i) {
150  s << x[i + 1] << ' ';
151  }
152  s << '\n';
153  double totaltime = 0;
154  for (size_t i = 0; i < seq_size - 1; ++i) {
155  s << "Leg time of flight: " << x[i + 4] << '\n';
156  totaltime += x[i + 4];
157  }
158  s << "Total time of flight: " << totaltime << '\n';
159  for (size_t i = 0; i < seq_size - 2; ++i) {
160  s << "Flyby radius: " << x[i + 2 * (seq_size + 1)] << '\n';
161  }
162  for (size_t i = 0; i < seq_size - 2; ++i) {
163  s << "Vinf at flyby: " << std::sqrt(problem.vrelin_vec[i]) << '\n';
164  }
165  for (size_t i = 0; i < seq_size - 1; ++i) {
166  s << "dsm" << i+1 << ": " << problem.DV[i+1] << '\n';
167  }

169 }
