363 // Helper function to insert a list of candidates immigrants into an immigrants vector, given the source and destination island.
364 void archipelago::build_immigrants_vector(std::vector<std::pair<population::size_type, individual_type > > &immigrants, const base_island &src_isl,
365  base_island &dest_isl, const std::vector<individual_type> &candidates) const
366 {
367  for (std::vector<individual_type>::const_iterator ind_it = candidates.begin();
368  ind_it != candidates.end(); ++ind_it)
369  {
370  // Skip individual if it is not within the bounds of the problem
371  // in the destination island.
372  if (!dest_isl.m_pop.problem().verify_x(ind_it->cur_x)) {
373  continue;
374  }
375  immigrants.push_back(std::make_pair(locate_island(src_isl),*ind_it));
376  }
377 }

379 // Re-evaluate vector of immigrants before insertion into destination island.
380 void archipelago::reevaluate_immigrants(std::vector<std::pair<population::size_type, individual_type> > &immigrants, const base_island &isl) const
381 {
382  individual_type tmp;
383  tmp.cur_v.resize(isl.m_pop.problem().get_dimension());
384  tmp.cur_f.resize(isl.m_pop.problem().get_f_dimension());
385  tmp.cur_c.resize(isl.m_pop.problem().get_c_dimension());
386  for (std::vector<std::pair<population::size_type, individual_type> >::iterator ind_it = immigrants.begin(); ind_it != immigrants.end(); ++ind_it) {
387  tmp.cur_x = (*ind_it).second.cur_x;
388  isl.m_pop.problem().objfun(tmp.cur_f,tmp.cur_x);
389  isl.m_pop.problem().compute_constraints(tmp.cur_c,tmp.cur_x);
390  // Set the best properties to the current ones. (TODO: maybe here one could
391  // reevaluate the old best in the new environment and keep it if still better than the
392  // reevaluated current ...... discuss!! (Anche no, grazie!!)
393  tmp.best_x = tmp.cur_x;
394  tmp.best_f = tmp.cur_f;
395  tmp.best_c = tmp.cur_c;
396  (*ind_it).second = tmp;
397  }
398 }

414 // This method will be called by each island of the archipelago before starting evolution. Its task is
415 // to select from the other islands, according to the topology and the migration/distribution type and direction,
416 // the individuals that will migrate into the island.
417 void archipelago::pre_evolution(base_island &isl)
418 {
419  // Make sure the island belongs to the archipelago.
420  pagmo_assert(isl.m_archi == this);
421  // Make sure the archipelago is not empty.
422  pagmo_assert(m_container.size());
423  // Determine the island's index in the archipelago.
424  const size_type isl_idx = locate_island(isl);
425  pagmo_assert(isl_idx < m_container.size());
426  //1. Obtain immigrants.
427  std::vector<std::pair<population::size_type, individual_type> > immigrants;
428  switch (m_migr_dir) {
429  case source:
430  {
431  lock_type lock(m_migr_mutex);
432  // For source migration direction, migration map contains islands' "inboxes". Or, in other words, it contains
433  // the individuals that are destined to go into the island. Such inboxes have been assembled previously,
434  // during a post_evolution operation.
435  // Iterate over all the vectors of individuals provided by the different islands.
436  for (boost::unordered_map<size_type,std::vector<individual_type> >::iterator it = m_migr_map[isl_idx].begin();
437  it != m_migr_map[isl_idx].end(); ++it)
438  {
439  pagmo_assert(it->first < m_container.size());
440  build_immigrants_vector(immigrants,*m_container[it->first],isl,it->second);
441  }
442  // Delete stuff in the migration map.
443  m_migr_map.erase(isl_idx);
444  break;
445  }
446  case destination:
447  // For destination migration direction, items in the migration map behave like "outboxes", i.e. each one is a
448  // "database of best individuals" seen in the islands of the archipelago.
449  // Get neighbours connecting into isl.
450  const std::vector<topology::base::vertices_size_type> inv_adj_islands(m_topology->get_v_inv_adjacent_vertices(boost::numeric_cast<topology::base::vertices_size_type>(isl_idx)));
451  // Do something only if there are adjacent islands.
452  if (inv_adj_islands.size()) {
453  switch (m_dist_type) {
454  case point_to_point:
455  {
456  lock_type lock(m_migr_mutex);
457  // Get the index of a random island connecting into isl.
458  boost::uniform_int<std::vector<topology::base::vertices_size_type>::size_type> u_int(0,inv_adj_islands.size() - 1);
459  const size_type rn_isl_idx = boost::numeric_cast<size_type>(inv_adj_islands[u_int(m_urng)]);
460  // Get the immigrants from the outbox of the random island. Note the redundant information in the last
461  // argument of the function.
462  pagmo_assert(m_migr_map[rn_isl_idx].size() <= 1);
463  build_immigrants_vector(immigrants,*m_container[rn_isl_idx],isl,m_migr_map[rn_isl_idx][rn_isl_idx]);
464  break;
465  }
466  case broadcast:
467  {
468  lock_type lock(m_migr_mutex);
469  // For broadcast migration fetch immigrants from all neighbour islands' databases.
470  for (std::vector<topology::base::vertices_size_type>::size_type i = 0; i < inv_adj_islands.size(); ++i) {
471  const size_type src_isl_idx = boost::numeric_cast<size_type>(inv_adj_islands[i]);
472  pagmo_assert(m_migr_map[src_isl_idx].size() <= 1);
473  build_immigrants_vector(immigrants,*m_container[src_isl_idx],isl,m_migr_map[src_isl_idx][src_isl_idx]);
474  }
475  }
476  }
477  }
478  }
479  //2. Insert immigrants into population.
480  if (immigrants.size()) {
481  double next_rng;
482  {
483  lock_type lock(m_migr_mutex);
484  next_rng = m_drng();
485  }
486  if (next_rng < isl.m_migr_prob) {
487  // We re-evaluate the incoming individuals according
488  // to destination island's problem. This will make sure that stochastic problems
489  // are correctly dealt with
490  reevaluate_immigrants(immigrants,isl);
491  // We then insert the incoming individuals into the population, storing how many from where
492  std::vector<std::pair<population::size_type, size_type> > rec_history;
493  rec_history = isl.accept_immigrants(immigrants);
494  lock_type lock(m_migr_mutex);
495  // Record the migration history.
496  for (size_t i =0; i< rec_history.size(); ++i) {
497  m_migr_hist.push_back( boost::make_tuple(
498  rec_history[i].first,
499  rec_history[i].second,
500  locate_island(isl) )
501  );
502  }
503  }
504  }
505 }

507 // This method will be called after each island isl has completed an evolution. Its purpose is to get individuals
508 // emigrating from isl and put them at disposal of the other islands of the archipelago, according to the migration attributes
509 // and the topology.
510 void archipelago::post_evolution(base_island &isl)
511 {
512  // Make sure the island belongs to the archipelago.
513  pagmo_assert(isl.m_archi == this);
514  // Make sure the archipelago is not empty.
515  pagmo_assert(m_container.size());
516  // Determine the island's index in the archipelago.
517  const size_type isl_idx = locate_island(isl);
518  pagmo_assert(isl_idx < m_container.size());
519  // Create the vector of emigrants.
520  std::vector<individual_type> emigrants;
521  switch (m_migr_dir) {
522  case source:
523  {
524  // Get the islands to which isl connects.
525  const std::vector<topology::base::vertices_size_type> adj_islands(m_topology->get_v_adjacent_vertices(boost::numeric_cast<topology::base::vertices_size_type>(isl_idx)));
526  if (adj_islands.size()) {
527  emigrants = isl.get_emigrants();
528  // Do something only if we have emigrants.
529  if (emigrants.size()) {
530  switch (m_dist_type)
531  {
532  case point_to_point:
533  {
534  lock_type lock(m_migr_mutex);
535  // For one-to-one migration choose a random neighbour island and put immigrants to its inbox.
536  boost::uniform_int<std::vector<topology::base::vertices_size_type>::size_type> u_int(0,adj_islands.size() - 1);
537  const size_type chosen_adj = boost::numeric_cast<size_type>(adj_islands[u_int(m_urng)]);
538  m_migr_map[chosen_adj][isl_idx].insert(m_migr_map[chosen_adj][isl_idx].end(),emigrants.begin(),emigrants.end());
539  break;
540  }
541  case broadcast:
542  {
543  lock_type lock(m_migr_mutex);
544  // For broadcast migration put immigrants to all neighbour islands' inboxes.
545  for (std::vector<topology::base::vertices_size_type>::size_type i = 0; i < adj_islands.size(); ++i) {
546  m_migr_map[boost::numeric_cast<size_type>(adj_islands[i])][isl_idx]
547  .insert(m_migr_map[boost::numeric_cast<size_type>(adj_islands[i])][isl_idx].end(),
548  emigrants.begin(),emigrants.end());
549  }
550  }
551  }
552  }
553  }
554  break;
555  }
556  case destination:
557  {
558  // For destination migration direction, migration map behaves like "outboxes", i.e. each is a "database of best individuals" for corresponding island.
559  emigrants = isl.get_emigrants();
560  lock_type lock(m_migr_mutex);
561  pagmo_assert(m_migr_map[isl_idx].size() <= 1);
562  m_migr_map[isl_idx][isl_idx].swap(emigrants);
563  }
564  }
565 }

594 void archipelago::evolve_batch(int n, unsigned int b)
595 {
596  join();
597  container_type::size_type arch_size = this->get_size();
598  // Variate generators
599  boost::uniform_int<int> pop_idx(0,arch_size-1);
600  boost::variate_generator<boost::mt19937 &, boost::uniform_int<int> > p_idx(m_urng,pop_idx);
601  // Shuffle to not bias towards low-indexes
602  std::vector<population::size_type> shuffle(arch_size);
603  for(population::size_type i=0; i < shuffle.size(); ++i) shuffle[i] = i;
604  std::random_shuffle(shuffle.begin(), shuffle.end(), p_idx);
605 
606  for(size_type p = 0; p < arch_size/b + 1; ++p) {
607  if(p == arch_size/b) { //for the last batch of islands decrease the barrier
608  reset_barrier(arch_size - p*b);
609  } else {
610  reset_barrier(b);
611  }
612  for(size_type i=0; i<b && p*b+i < arch_size; ++i) {
613  m_container[shuffle[p*b+i]]->evolve(n);
614  }
615  for(size_type i=0; i<b && p*b+i < arch_size; ++i) {
616  m_container[shuffle[p*b+i]]->join();
617  }
618  }
619 }
