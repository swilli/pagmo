42 mga_target_event::mga_target_event(
43  const kep_toolbox::planet_ptr start,
44  const kep_toolbox::planet_ptr end,
45  const kep_toolbox::epoch t_end,
46  double T_max,
47  bool discount_launcher
48  ):base(6), m_start(start), m_end(end), m_t_end(t_end), m_T_max(T_max), m_discount_launcher(discount_launcher)
49 {
50  // We check that all planets have equal central body
51  if (start->get_mu_central_body() != end->get_mu_central_body()) {
52  pagmo_throw(value_error,"The planets do not all have the same mu_central_body");
53  }
54 
55  // We check that T_max is positive
56  if (T_max <= 0) {
57  pagmo_throw(value_error,"T_max must be larger than zero");
58  }
59 
60  // Now setting the problem bounds
61  decision_vector lb(6,0.0), ub(6,1.0);
62  lb[0] = 1.; lb[5] = 1e-5;
63  ub[0] = T_max; ub[2] = 12000.0; ub[5] = 1-1e-5;
64 
65  set_bounds(lb,ub);
66 }

83 void mga_target_event::objfun_impl(fitness_vector &f, const decision_vector &x) const
84 {
85  //1 - we 'decode' the chromosome
86  double tw = x[0]*x[1]; //transforms the waiting time from a fraction to days
87 
88  //2 - We compute the epochs and ephemerides of the planetary encounters
89  kep_toolbox::epoch t0 = kep_toolbox::epoch(m_t_end.mjd2000() - x[0]); //epoch of departure
90  double tof = (x[0] - tw); //in days
91 //std::cout << "tof: " << tof << std::endl;
92  kep_toolbox::epoch t1 = kep_toolbox::epoch(t0.mjd2000() + tof); //epoch
93 
94  kep_toolbox::array3D r0,v0,r1,v1,v_inf_vett,v0_sc;
95  m_start->get_eph(t0,r0,v0);
96  m_end->get_eph(t1,r1,v1);
97 //std::cout << "r0: " << r0 << std::endl;
98 //std::cout << "v0: " << v0 << std::endl;
99 //std::cout << "r1: " << r1 << std::endl;
100 //std::cout << "v1: " << v1 << std::endl;
101 
102  //3 - We compute the absolute velocity at departure
103  double theta = 2*M_PI*x[3];
104  double phi = acos(2. * x[4]-1.) - M_PI / 2.;
105  v_inf_vett[0] = x[2]*cos(phi)*cos(theta);
106  v_inf_vett[1] = x[2]*cos(phi)*sin(theta);
107  v_inf_vett[2] = x[2]*sin(phi);
108  kep_toolbox::sum(v0_sc,v0,v_inf_vett);
109 
110 //std::cout << "v_inf_vett: " << v_inf_vett << std::endl;
111 //std::cout << "v_sc: " << v0_sc << std::endl;
112 //std::cout << "tof: " << tof << std::endl;
113 //std::cout << "x[5]: " << x[5] << std::endl;
114 //std::cout << "x: " << x << std::endl;
115 
116  //4 - And we propagate up to the DSM positon to then solve a Lambert problem
117  kep_toolbox::propagate_lagrangian(r0,v0_sc,tof*x[5]*ASTRO_DAY2SEC, ASTRO_MU_SUN);
118  kep_toolbox::lambert_problem l(r0,r1,tof*(1-x[5])*ASTRO_DAY2SEC,ASTRO_MU_SUN, false,0);
119 
120 //std::cout << "r0: " << r0 << std::endl;
121 //std::cout << "v0: " << v0_sc << std::endl;
122 
123  double DV1 = x[2];
124  kep_toolbox::array3D dv2,dv3;
125  kep_toolbox::diff(dv2,v0_sc,l.get_v1()[0]);
126  double DV2 = kep_toolbox::norm(dv2);
127  kep_toolbox::diff (dv3,v1,l.get_v2()[0]);
128  double DV3 = kep_toolbox::norm(dv3);
129  if (m_discount_launcher) {
130  DV1 = std::max(0.,DV1-6000.);
131  }
132  f[0] = DV1+DV2+DV3;
133 // std::cout << "DV1: " << DV1 << std::endl;
134 // std::cout << "DV2: " << DV2 << std::endl;
135 // std::cout << "DV3: " << DV3 << std::endl;
136 }
