128 void hypervolume::verify_after_construct() const
129 {
130  if ( m_points.size() == 0 ) {
131  pagmo_throw(value_error, "Point set cannot be empty.");
132  }
133  fitness_vector::size_type f_dim = m_points[0].size();
134  if (f_dim <= 1) {
135  pagmo_throw(value_error, "Points of dimension > 1 required.");
136  }
137  for (std::vector<fitness_vector>::size_type idx = 1 ; idx < m_points.size() ; ++idx) {
138  if ( m_points[idx].size() != f_dim ) {
139  pagmo_throw(value_error, "All point set dimensions must be equal.");
140  }
141  }
142 }

439 fitness_vector hypervolume::get_nadir_point(const double epsilon) const
440 {
441  fitness_vector nadir_point(m_points[0].begin(), m_points[0].end());
442  for (std::vector<fitness_vector>::size_type idx = 1 ; idx < m_points.size() ; ++ idx){
443  for (fitness_vector::size_type f_idx = 0 ; f_idx < m_points[0].size() ; ++f_idx){
444  // assuming minimization problem, thus maximum value by each dimension is taken
445  nadir_point[f_idx] = std::max(nadir_point[f_idx], m_points[idx][f_idx]);
446  }
447  }
448  for (fitness_vector::size_type f_idx = 0 ; f_idx < nadir_point.size() ; ++f_idx) {
449  nadir_point[f_idx] += epsilon;
450  }
451  return nadir_point;
452 }
