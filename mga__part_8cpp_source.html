52 mga_part::mga_part(const std::vector<kep_toolbox::planet_ptr> seq,
53  const std::vector<std::vector<double> > tof,
54  const kep_toolbox::epoch t0,
55  const kep_toolbox::array3D vinf_in
56  ):
57  base(4*(seq.size()-1)), m_tof(tof), m_t0(t0), m_vinf_in(vinf_in)
58 {
59  // We check that all planets have equal central body
60  std::vector<double> mus(seq.size());
61  for (std::vector<kep_toolbox::planet_ptr>::size_type i = 0; i< seq.size(); ++i) {
62  mus[i] = seq[i]->get_mu_central_body();
63  }
64  if ((unsigned int)std::count(mus.begin(), mus.end(), mus[0]) != mus.size()) {
65  pagmo_throw(value_error,"The planets do not all have the same mu_central_body");
66  }
67  // We check the consistency of the time of flights
68  if (tof.size() != (seq.size()-1)) {
69  pagmo_throw(value_error,"The time-of-flight vector (tof) has the wrong length");
70  }
71  for (size_t i = 0; i < tof.size(); ++i) {
72  if (tof[i].size()!=2) pagmo_throw(value_error,"Each element of the time-of-flight vector (tof) needs to have dimension 2 (lower and upper bound)");
73  }
74  //We check the sequence contains at least two planets
75  if (seq.size() < 2) {
76  pagmo_throw(value_error,"The sequence must contain at least 2 planets");
77  }
78 
79  // Filling in the planetary sequence data member. This requires to construct the polymorphic planets via their clone method
80  for (std::vector<kep_toolbox::planet>::size_type i = 0; i < seq.size(); ++i) {
81  m_seq.push_back(seq[i]->clone());
82  }
83 
84  // Now setting the problem bounds
85  size_type dim(4*(seq.size()-1));
86  decision_vector lb(dim), ub(dim);
87 
88 
89  // all legs
90  for (std::vector<kep_toolbox::planet>::size_type i = 0; i < (m_seq.size() - 1); ++i) {
91  lb[4*i] = - 2 * boost::math::constants::pi<double>(); ub[4*i] = 2 * boost::math::constants::pi<double>();
92  lb[4*i+1] = 1.1; ub[4*i+1] = 30;
93  lb[4*i+2] = 1e-5; ub[4*i+2] = 1-1e-5;
94  lb[4*i+3] = m_tof[i][0]; ub[4*i+3] = m_tof[i][1];
95  }
96 
97  // Adjusting the minimum and maximum allowed fly-by rp to the one defined in the kep_toolbox::planet class
98  for (std::vector<kep_toolbox::planet>::size_type i = 0; i < (m_seq.size() - 1); ++i) {
99  lb[4*i+1] = m_seq[i]->get_safe_radius() / m_seq[i]->get_radius();
100  ub[4*i+1] = (m_seq[i]->get_radius() + 2000000) / m_seq[i]->get_radius(); //from gtoc6 problem description
101  }
102  set_bounds(lb,ub);
103 }

121 void mga_part::objfun_impl(fitness_vector &f, const decision_vector &x) const
122 {
123 try {
124  double d,ra,d2,ra2;
125  double common_mu = m_seq[0]->get_mu_central_body();
126  // 1 - we 'decode' the chromosome recording the various times of flight (days) in the list T
127  std::vector<double> T(m_seq.size()-1,0.0);
128 
129  for (size_t i = 0; i < (m_seq.size()-1); ++i) {
130  T[i] = x[4*i+3];
131  }
132  // 2 - We compute the epochs and ephemerides of the planetary encounters
133  std::vector<kep_toolbox::epoch> t_P(m_seq.size());
134  std::vector<kep_toolbox::array3D> r_P(m_seq.size());
135  std::vector<kep_toolbox::array3D> v_P(m_seq.size());
136  std::vector<double> DV(m_seq.size() - 1);
137  for (size_t i = 0; i<m_seq.size(); ++i) {
138  t_P[i] = kep_toolbox::epoch(m_t0.mjd2000() + std::accumulate(T.begin(),T.begin()+i,0.0));
139  m_seq[i]->get_eph(t_P[i], r_P[i], v_P[i]);
140  }
141 
142  // 3 - We loop over the legs
143  kep_toolbox::array3D v_out,r,v;
144  kep_toolbox::array3D v_beg_l,v_end_l;
145  kep_toolbox::sum(v_end_l, m_vinf_in, v_P[0]);
146  for (size_t i = 0; i<(m_seq.size()-1); ++i) {
147  // Fly-by
148  kep_toolbox::fb_prop(v_out, v_end_l, v_P[i], x[4*i+1] * m_seq[i]->get_radius(), x[4*i], m_seq[i]->get_mu_self());
149  // s/c propagation before the DSM
150  r = r_P[i];
151  v = v_out;
152 
153  kep_toolbox::propagate_lagrangian(r,v,x[4*i+2]*T[i]*ASTRO_DAY2SEC,common_mu);
154  kep_toolbox::closest_distance(d, ra, r_P[i], v_out, r, v, common_mu);
155 
156  // Lambert arc to reach Earth during (1-nu2)*T2 (second segment)
157  double dt = (1-x[4*i+2])*T[i]*ASTRO_DAY2SEC;
158  kep_toolbox::lambert_problem l2(r,r_P[i+1],dt,common_mu,false,false);
159  v_end_l = l2.get_v2()[0];
160  v_beg_l = l2.get_v1()[0];
161  kep_toolbox::closest_distance(d2,ra2,r,v_beg_l, r_P[i+1], v_end_l, common_mu);
162 
163  if (d < d2)
164  {
165  d = d/ASTRO_JR;
166  } else {
167  d = d2/ASTRO_JR;
168  }
169 
170  // DSM occuring at time nu2*T2
171  kep_toolbox::diff(v_out, v_beg_l, v);
172  // Penalized if the JR is too small (<0.2)
173  DV[i] = kep_toolbox::norm(v_out) + std::max((2.0-d),0.0) * 1000.0;
174  }
175  // Now we return the objective(s) function
176  f[0] = std::accumulate(DV.begin(),DV.end(),0.0);
177 //Here the lambert solver or the lagrangian propagator went wrong
178 } catch (...) {
179  f[0] = boost::numeric::bounds<double>::highest();
180 }
181 }
