88  void tsp_ds::objfun_impl(fitness_vector &f, const decision_vector& x) const
89  {
90  f[0]=0;
91  decision_vector tour;
92  decision_vector::size_type n_cities = get_n_cities();
93  double cum_p, ham_path_len;
94  decision_vector::size_type dumb1, dumb2;
95 
96  switch( get_encoding() ) {
97  case FULL:
98  {
99  tour = full2cities(x);
100  find_subsequence(tour, cum_p, ham_path_len, dumb1, dumb2);
101  break;
102  }
103  case RANDOMKEYS:
104  {
105  tour = randomkeys2cities(x);
106  find_subsequence(tour, cum_p, ham_path_len, dumb1, dumb2);
107  break;
108  }
109  case CITIES:
110  {
111  find_subsequence(x, cum_p, ham_path_len, dumb1, dumb2);
112  }
113  }
114  ham_path_len = std::accumulate(m_DV.begin(), m_DV.end(), 0.0);
115  f[0] = -(cum_p + 1 - ham_path_len / (n_cities * m_max_DV * 10));
116  return;
117  }

133  void tsp_ds::find_subsequence(const decision_vector& tour, double& retval_p, double& retval_l, decision_vector::size_type& retval_it_l, decision_vector::size_type& retval_it_r, const bool static_computations) const
134  {
135  if (tour.size() != get_n_cities())
136  {
137  pagmo_throw(value_error, "tour dimension must be equal to the number of planets");
138  }
139 
140  // We declare the necessary variable
141  decision_vector::size_type n_cities = get_n_cities();
142  decision_vector::size_type it_l = 0, it_r = 0;
143  bool cond_r = true, cond_l = true;
144  double cum_p = m_values[tour[0]];
145  double saved_length = m_max_DV;
146 
147  // We initialize the starting values
148  retval_p = cum_p;
149  retval_l = saved_length;
150  retval_it_l = it_l;
151  retval_it_r = it_r;
152 
153  // We precompute all DVs (stored in m_DV)
154  compute_DVs(tour, static_computations);
155 
156  // Main body of the double loop
157  while(cond_l)
158  {
159  while(cond_r)
160  {
161  // We increment the right "pointer" updating the value and length of the path
162  saved_length -= m_DV[it_r];
163  cum_p += m_values[tour[(it_r + 1)]];
164  it_r += 1;
165 
166  // We update the various retvals only if the new subpath is valid
167  if (saved_length < 0 || (it_l == it_r))
168  {
169  cond_r = false;
170  }
171  else if (cum_p > retval_p)
172  {
173  retval_p = cum_p;
174  retval_l = saved_length;
175  retval_it_l = it_l;
176  retval_it_r = it_r;
177  }
178  else if (cum_p == retval_p)
179  {
180  if (saved_length > retval_l)
181  {
182  retval_p = cum_p;
183  retval_l = saved_length;
184  retval_it_l = it_l;
185  retval_it_r = it_r;
186  }
187  }
188 
189  if (it_r==n_cities-1)
190  {
191  goto EndOfLoop;
192  }
193  }
194  // We get out if all cities are included in the current path
195  if (it_l == it_r)
196  {
197  cond_l = false;
198  }
199  else
200  {
201  // We increment the left "pointer" updating the value and length of the path
202  saved_length += m_DV[it_l];
203  cum_p -= m_values[tour[it_l]];
204  it_l += 1;
205  // We update the various retvals only if the new subpath is valid
206  if (saved_length > 0)
207  {
208  cond_r = true;
209  if (cum_p > retval_p)
210  {
211  retval_p = cum_p;
212  retval_l = saved_length;
213  retval_it_l = it_l;
214  retval_it_r = it_r;
215  }
216  else if (cum_p == retval_p)
217  {
218  if (saved_length > retval_l)
219  {
220  retval_p = cum_p;
221  retval_l = saved_length;
222  retval_it_l = it_l;
223  retval_it_r = it_r;
224  }
225  }
226  }
227  if (it_l == n_cities)
228  {
229  cond_l = false;
230  }
231  }
232  }
233 EndOfLoop:
234  return;
235  }

243  void tsp_ds::compute_constraints_impl(constraint_vector &c, const decision_vector& x) const
244  {
245  decision_vector::size_type n_cities = get_n_cities();
246 
247  switch( get_encoding() )
248  {
249  case FULL:
250  {
251  // 1 - We set the equality constraints
252  for (size_t i = 0; i < n_cities; i++) {
253  c[i] = 0;
254  c[i+n_cities] = 0;
255  for (size_t j = 0; j < n_cities; j++) {
256  if(i==j) continue; // ignoring main diagonal
257  decision_vector::size_type rows = compute_idx(i, j, n_cities);
258  decision_vector::size_type cols = compute_idx(j, i, n_cities);
259  c[i] += x[rows];
260  c[i+n_cities] += x[cols];
261  }
262  c[i] = c[i]-1;
263  c[i+n_cities] = c[i+n_cities]-1;
264  }
265 
266  //2 - We set the inequality constraints
267  //2.1 - First we compute the uj (see http://en.wikipedia.org/wiki/Travelling_salesman_problem#Integer_linear_programming_formulation)
268  // we start always out tour from the first city, without loosing generality
269  size_t next_city = 0,current_city = 0;
270  std::vector<int> u(n_cities);
271  for (size_t i = 0; i < n_cities; i++) {
272  u[current_city] = i+1;
273  for (size_t j = 0; j < n_cities; j++)
274  {
275  if (current_city==j) continue;
276  if (x[compute_idx(current_city, j, n_cities)] == 1)
277  {
278  next_city = j;
279  break;
280  }
281  }
282  current_city = next_city;
283  }
284  int count=0;
285  for (size_t i = 1; i < n_cities; i++) {
286  for (size_t j = 1; j < n_cities; j++)
287  {
288  if (i==j) continue;
289  c[2*n_cities+count] = u[i]-u[j] + (n_cities+1) * x[compute_idx(i, j, n_cities)] - n_cities;
290  count++;
291  }
292  }
293  break;
294  }
295  case RANDOMKEYS:
296  break;
297  case CITIES:
298  {
299  std::vector<population::size_type> range(n_cities);
300  for (std::vector<population::size_type>::size_type i=0; i<range.size(); ++i)
301  {
302  range[i]=i;
303  }
304  c[0] = !std::is_permutation(x.begin(),x.end(),range.begin());
305  break;
306  }
307  }
308  return;
309  }

405  double tsp_ds::three_impulses(double a1, double i1, double W1, double e1, double a2, double i2, double W2, double e2) const
406  {
407  using namespace std;
408  double Vi,Vf, DV1,DV2;
409 
410  // radius of apocenter starting orbit (km)
411  double ra1 = a1 * (1. + e1);
412  // radius of apocenter target orbit(km)
413  double ra2 = a2 * (1. + e2);
414  // relative inclination between orbits
415  double cosiREL = cos(i1) * cos(i2) + sin(i1) * sin(i2) * cos(W1) * cos(W2) + sin(i1) * sin(i2) * sin(W1) * sin(W2);
416 
417  // radius of pericenter target orbit(km)
418  double rp2 = a2 * (1. - e2);
419  // radius of pericenter starting orbit (km)
420  double rp1 = a1 * (1. - e1);
421 
422  if (ra1 > ra2) { // Strategy is Apocenter-Pericenter
423  Vi = sqrt(m_mu * (2. / ra1 - 1. / a1));
424  Vf = sqrt(m_mu * (2. / ra1 - 2. / (rp2 + ra1)));
425  // Change Inclination + pericenter change
426  DV1 = sqrt(Vi * Vi + Vf * Vf - 2. * Vi * Vf * cosiREL);
427  // Apocenter Change
428  DV2 = sqrt(m_mu) * abs(sqrt(2. / rp2 - 2. / (rp2 + ra1)) - sqrt(2. / rp2 - 2. / (rp2 + ra2)));
429  }
430  else { // (ra1<ra2) Strategy is Pericenter-Apocenter
431  // Apocenter Raise
432  DV1 = sqrt(m_mu) * abs(sqrt(2. / rp1 - 2. / (rp1 + ra1)) - sqrt(2. / rp1 - 2. / (rp1 + ra2)));
433  Vi = sqrt(m_mu * (2. / ra2 - 2. / (rp1 + ra2)));
434  Vf = sqrt(m_mu * (2. / ra2 - 1. / a2));
435  // Change Inclination + apocenter change
436  DV2 = sqrt(abs(Vi * Vi + Vf * Vf - 2 * Vi * Vf * cosiREL));
437  }
438  return DV1 + DV2;
439  }
