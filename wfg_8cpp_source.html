87 std::vector<double> wfg::contributions(std::vector<fitness_vector> &points, const fitness_vector &r_point) const
88 {
89  std::vector<double> c;
90  c.reserve(points.size());
91 
92  // Allocate the same members as for 'compute' method
93  allocate_wfg_members(points, r_point);
94 
95  // Prepare the memory for first front
96  double** fr = new double*[m_max_points];
97  for(unsigned int i = 0 ; i < m_max_points ; ++i) {
98  fr[i]=new double[m_current_slice];
99  }
100  m_frames[m_n_frames] = fr;
101  m_frames_size[m_n_frames] = 0;
102  ++m_n_frames;
103 
104  for(unsigned int p_idx = 0 ; p_idx < m_max_points ; ++p_idx) {
105  limitset(0, p_idx, 1);
106  c.push_back(exclusive_hv(p_idx, 1));
107  }
108 
109  // Free the contributions and the remaining WFG members
110  free_wfg_members();
111 
112  return c;

116 void wfg::allocate_wfg_members(std::vector<fitness_vector> &points, const fitness_vector &r_point) const
117 {
118  m_max_points = points.size();
119  m_max_dim = r_point.size();
120 
121  m_refpoint = new double[m_max_dim];
122  for(unsigned int d_idx = 0 ; d_idx < m_max_dim ; ++d_idx) {
123  m_refpoint[d_idx] = r_point[d_idx];
124  }
125 
126  // Reserve the space beforehand for each level or recursion.
127  // WFG with slicing feature will not go recursively deeper than the dimension size.
128  m_frames = new double**[m_max_dim];
129  m_frames_size = new unsigned int[m_max_dim];
130 
131  // Copy the initial set into the frame at index 0.
132  double** fr = new double*[m_max_points];
133  for(unsigned int p_idx = 0 ; p_idx < m_max_points ; ++p_idx) {
134  fr[p_idx] = new double[m_max_dim];
135  for(unsigned int d_idx = 0 ; d_idx < m_max_dim ; ++d_idx) {
136  fr[p_idx][d_idx] = points[p_idx][d_idx];
137  }
138  }
139  m_frames[0] = fr;
140  m_frames_size[0] = m_max_points;
141  m_n_frames = 1;
142 
143  // Variable holding the current "depth" of dimension slicing. We progress by slicing dimensions from the end.
144  m_current_slice = m_max_dim;
145 }

149 {

164 void wfg::limitset(const unsigned int begin_idx, const unsigned int p_idx, const unsigned int rec_level) const
165 {
166  double **points = m_frames[rec_level - 1];
167  unsigned int n_points = m_frames_size[rec_level - 1];
168 
169  int no_points = 0;
170 
171  double* p = points[p_idx];
172  double** frame = m_frames[rec_level];
173 
174  for(unsigned int idx = begin_idx; idx < n_points; ++idx) {
175  if (idx == p_idx) {
176  continue;
177  }
178 
179  for(fitness_vector::size_type f_idx = 0; f_idx < m_current_slice; ++f_idx) {
180  frame[no_points][f_idx] = std::max(points[idx][f_idx], p[f_idx]);
181  }
182 
183  std::vector<int> cmp_results;
184  cmp_results.resize(no_points);
185  double* s = frame[no_points];
186 
187  bool keep_s = true;
188 
189  // Check whether any point is dominating the point 's'.
190  for(int q_idx = 0; q_idx < no_points; ++q_idx) {
191  cmp_results[q_idx] = base::dom_cmp(s, frame[q_idx], m_current_slice);
192  if (cmp_results[q_idx] == base::DOM_CMP_B_DOMINATES_A) {
193  keep_s = false;
194  break;
195  }
196  }
197  // If neither is, remove points dominated by 's' (we store that during the first loop).
198  if( keep_s ) {
199  int prev = 0;
200  int next = 0;
201  while(next < no_points) {
202  if( cmp_results[next] != base::DOM_CMP_A_DOMINATES_B && cmp_results[next] != base::DOM_CMP_A_B_EQUAL) {
203  if(prev < next) {
204  for(unsigned int d_idx = 0; d_idx < m_current_slice ; ++d_idx) {
205  frame[prev][d_idx] = frame[next][d_idx];
206  }
207  }
208  ++prev;
209  }
210  ++next;
211  }
212  // Append 's' at the end, if prev==next it's not necessary as it's already there.
213  if(prev < next) {
214  for(unsigned int d_idx = 0; d_idx < m_current_slice ; ++d_idx) {
215  frame[prev][d_idx] = s[d_idx];
216  }
217  }
218  no_points = prev + 1;
219  }
220  }
221 
222  m_frames_size[rec_level] = no_points;
223 }

226 double wfg::compute_hv(const unsigned int rec_level) const
227 {
228  double **points = m_frames[rec_level - 1];
229  unsigned int n_points = m_frames_size[rec_level - 1];
230 
231  // Simple inclusion-exclusion for one and two points
232  if (n_points == 1) {
233  return base::volume_between(points[0], m_refpoint, m_current_slice);
234  }
235  else if (n_points == 2) {
236  double hv = base::volume_between(points[0], m_refpoint, m_current_slice)
237  + base::volume_between(points[1], m_refpoint, m_current_slice);
238  double isect = 1.0;
239  for(unsigned int i=0;i<m_current_slice;++i) {
240  isect *= (m_refpoint[i] - std::max(points[0][i], points[1][i]));
241  }
242  return hv - isect;
243  }
244 
245  // If already sliced to dimension at which we use another algorithm.
246  if (m_current_slice == m_stop_dimension) {
247 
248  if (m_stop_dimension == 2) {
249  // Use a very efficient version of hv2d
250  return hv2d().compute(points, n_points, m_refpoint);
251  } else {
252  // Let hypervolume object pick the best method otherwise.
253  std::vector<fitness_vector> points_cpy;
254  points_cpy.reserve(n_points);
255  for(unsigned int i = 0 ; i < n_points ; ++i) {
256  points_cpy.push_back(fitness_vector(points[i], points[i] + m_current_slice));
257  }
258  fitness_vector r_cpy(m_refpoint, m_refpoint + m_current_slice);
259 
260  hypervolume hv = hypervolume(points_cpy, false);
261  hv.set_copy_points(false);
262  return hv.compute(r_cpy);
263  }
264  } else {
265  // Otherwise, sort the points in preparation for the next recursive step
266  // Bind the object under "this" pointer to the cmp_points method so it can be used as a valid comparator function for std::sort
267  // We need that in order for the cmp_points to have acces to the m_current_slice member variable.
268  std::sort(points, points + n_points, boost::bind(&wfg::cmp_points, this, _1, _2));
269  }
270 
271  double H = 0.0;
272  --m_current_slice;
273 
274  if(rec_level >= m_n_frames) {
275  double** fr = new double*[m_max_points];
276  for(unsigned int i = 0 ; i < m_max_points ; ++i) {
277  fr[i]=new double[m_current_slice];
278  }
279  m_frames[m_n_frames] = fr;
280  m_frames_size[m_n_frames] = 0;
281  ++m_n_frames;
282  }
283 
284  for(unsigned int p_idx = 0 ; p_idx < n_points ; ++p_idx) {
285  limitset(p_idx + 1, p_idx, rec_level);
286 
287  H += fabs((points[p_idx][m_current_slice] - m_refpoint[m_current_slice]) * exclusive_hv(p_idx, rec_level));
288  }
289  ++m_current_slice;
290  return H;
291 }

294 double wfg::exclusive_hv(const unsigned int p_idx, const unsigned int rec_level) const
295 {
296  //double H = base::volume_between(points[p_idx], m_refpoint, m_current_slice);
297  double H = base::volume_between(m_frames[rec_level - 1][p_idx], m_refpoint, m_current_slice);
298 
299  if (m_frames_size[rec_level] == 1) {
300  H -= base::volume_between(m_frames[rec_level][0], m_refpoint, m_current_slice);
301  } else if (m_frames_size[rec_level] > 1) {
302  H -= compute_hv(rec_level + 1);
303  }
304 
305  return H;
306 }
