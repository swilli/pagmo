77 void sample_return::objfun_impl(fitness_vector &f, const decision_vector &x) const
78 {
79  //We split the decision vector in the two legs
80  std::copy(x.begin(),x.begin()+6,x_leg1.begin());
81  std::copy(x.begin()+6,x.begin()+12,x_leg2.begin());
82 
83  x_leg1[4] = x_leg1[4] * m_Tmax;
84  x_leg2[4] = (m_Tmax - x_leg1[4] - x_leg2[0]) * x_leg2[4];
85 
86  //We account for the waiting time
87  x_leg2[0] += x_leg1[0] + x_leg1[4];
88  double dummy = 0;
89  MGA_DSM(x_leg1, m_leg1, dummy);
90  MGA_DSM(x_leg2, m_leg2, dummy);
91  f[0] = m_leg1.DV[0] + m_leg1.DV[1] + m_leg1.DV[2] +
92  m_leg2.DV[0] + m_leg2.DV[1] + std::max(0.0,m_leg2.DV[2] - 5.5);
93 
94 }
