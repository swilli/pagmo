52 mga_incipit_cstrs::mga_incipit_cstrs(
53  const std::vector<kep_toolbox::planet_ptr> seq,
54  const kep_toolbox::epoch t0_l,
55  const kep_toolbox::epoch t0_u,
56  const std::vector<std::vector<double> > tof,
57  double Tmax,
58  double Dmin) : base(4*seq.size(),0,1,2,2,1E-3), m_tof(tof), m_tmax(Tmax), m_dmin(Dmin)
59 {
60  // We check that all planets have equal central body
61  std::vector<double> mus(seq.size());
62  for (std::vector<kep_toolbox::planet_ptr>::size_type i = 0; i< seq.size(); ++i) {
63  mus[i] = seq[i]->get_mu_central_body();
64  }
65  if ((unsigned int)std::count(mus.begin(), mus.end(), mus[0]) != mus.size()) {
66  pagmo_throw(value_error,"The planets do not all have the same mu_central_body");
67  }
68  // We check the consistency of the time of flights
69  if (tof.size() != seq.size()) {
70  pagmo_throw(value_error,"The time-of-flight vector (tof) has the wrong length");
71  }
72  for (size_t i = 0; i < tof.size(); ++i) {
73  if (tof[i].size()!=2) pagmo_throw(value_error,"Each element of the time-of-flight vector (tof) needs to have dimension 2 (lower and upper bound)");
74  }
75 
76  // Filling in the planetary sequence data member. This requires to construct the polymorphic planets via their clone method
77  for (std::vector<kep_toolbox::planet>::size_type i = 0; i < seq.size(); ++i) {
78  m_seq.push_back(seq[i]->clone());
79  }
80 
81  // Now setting the problem bounds
82  size_type dim(4*m_tof.size());
83  decision_vector lb(dim), ub(dim);
84 
85  // First leg
86  lb[0] = t0_l.mjd2000(); ub[0] = t0_u.mjd2000();
87  lb[1] = 0; lb[2] = 0; ub[1] = 1; ub[2] = 1;
88  lb[3] = m_tof[0][0]; ub[3] = m_tof[0][1];
89 
90  // Successive legs
91  for (std::vector<kep_toolbox::planet>::size_type i = 1; i < m_tof.size(); ++i) {
92  lb[4*i] = - 2 * boost::math::constants::pi<double>(); ub[4*i] = 2 * boost::math::constants::pi<double>();
93  lb[4*i+1] = 1.1; ub[4*i+1] = 30;
94  lb[4*i+2] = 1e-5; ub[4*i+2] = 1-1e-5;
95  lb[4*i+3] = m_tof[i][0]; ub[4*i+3] = m_tof[i][1];
96  }
97 
98  // Adjusting the minimum and maximum allowed fly-by rp to the one defined in the kep_toolbox::planet class
99  for (std::vector<kep_toolbox::planet>::size_type i = 0; i < m_tof.size()-1; ++i) {
100  lb[4*i+5] = m_seq[i]->get_safe_radius() / m_seq[i]->get_radius();
101  ub[4*i+5] = (m_seq[i]->get_radius() + 2000000) / m_seq[i]->get_radius(); //from gtoc6 problem description
102  }
103  set_bounds(lb,ub);
104 }

126 void mga_incipit_cstrs::objfun_impl(fitness_vector &f, const decision_vector &x) const
127 {
128 try {
129  double common_mu = m_seq[0]->get_mu_central_body();
130  // 1 - we 'decode' the chromosome recording the various times of flight (days) in the list T
131  std::vector<double> T(m_seq.size(),0.0);
132 
133  for (size_t i = 0; i<m_seq.size(); ++i) {
134  T[i] = x[4*i+3];
135  }
136  // 2 - We compute the epochs and ephemerides of the planetary encounters
137  std::vector<kep_toolbox::epoch> t_P(m_seq.size());
138  std::vector<kep_toolbox::array3D> r_P(m_seq.size());
139  std::vector<kep_toolbox::array3D> v_P(m_seq.size());
140  std::vector<double> DV(m_seq.size());
141  for (size_t i = 0; i<r_P.size(); ++i) {
142  t_P[i] = kep_toolbox::epoch(x[0] + std::accumulate(T.begin(),T.begin()+1+i,0.0));
143  m_seq[i]->get_eph(t_P[i], r_P[i], v_P[i]);
144  }
145 
146  // 3 - We start with the first leg
147  double theta = 2*boost::math::constants::pi<double>()*x[1];
148  double phi = acos(2*x[2]-1)-boost::math::constants::pi<double>() / 2;
149  double d,d2,ra,ra2;
150  kep_toolbox::array3D r = { {ASTRO_JR*1000*cos(phi)*sin(theta), ASTRO_JR*1000*cos(phi)*cos(theta), ASTRO_JR*1000*sin(phi)} };
151  kep_toolbox::array3D v;
152  kep_toolbox::lambert_problem l(r,r_P[0],T[0]*ASTRO_DAY2SEC,common_mu,false,false);
153  kep_toolbox::array3D v_beg_l = l.get_v1()[0];
154  kep_toolbox::array3D v_end_l = l.get_v2()[0];
155 
156  DV[0] = std::abs(kep_toolbox::norm(v_beg_l)-3400.0);
157 
158  // 4 - And we proceed with each successive leg (if any)
159  kep_toolbox::array3D v_out;
160  for (size_t i = 1; i<m_seq.size(); ++i) {
161  // Fly-by
162  kep_toolbox::fb_prop(v_out, v_end_l, v_P[i-1], x[4*i+1] * m_seq[i-1]->get_radius(), x[4*i], m_seq[i-1]->get_mu_self());
163  r = r_P[i-1];
164  v = v_out;
165  // s/c propagation before the DSM
166  kep_toolbox::propagate_lagrangian(r,v,x[4*i+2]*T[i]*ASTRO_DAY2SEC,common_mu);
167  kep_toolbox::closest_distance(d, ra, r_P[i-1], v_out, r, v, common_mu);
168 
169  // Lambert arc to reach Earth during (1-nu2)*T2 (second segment)
170  double dt = (1-x[4*i+2])*T[i]*ASTRO_DAY2SEC;
171  kep_toolbox::lambert_problem l2(r,r_P[i],dt,common_mu,false,false);
172  v_end_l = l2.get_v2()[0];
173  v_beg_l = l2.get_v1()[0];
174  kep_toolbox::closest_distance(d2,ra2,r,v_beg_l, r_P[i], v_end_l, common_mu);
175  if (d < d2)
176  {
177  d = d/ASTRO_JR;
178  } else {
179  d = d2/ASTRO_JR;
180  }
181 
182  // DSM occuring at time nu2*T2
183  kep_toolbox::diff(v_out, v_beg_l, v);
184  DV[i] = kep_toolbox::norm(v_out) + std::max((2.0-d),0.0) * 1000.0;
185  }
186  // Now we return the objective(s) function
187  f[0] = std::accumulate(DV.begin(),DV.end(),0.0);
188 //Here the lambert solver or the lagrangian propagator went wrong
189 } catch (...) {
190  f[0] = boost::numeric::bounds<double>::highest();
191 }
